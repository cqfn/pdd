def xml
  Nokogiri::XML::Builder.new do |xml|
    xml << '<?xml-stylesheet type="text/xsl" href="puzzles.xsl"?>'
    xml.puzzles do
      Sources.new(@dir).fetch.each do |source|
        source.puzzles.each { |puzzle| render puzzle, xml }
      end
    end
  end.to_xml
end
